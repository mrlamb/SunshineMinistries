CREATE PROCEDURE [dbo].[ContactAddOrEdit]
	@mode nvarchar(10),
	@ContactId int,
	@FirstName nvarchar(50),
	@LastName nvarchar(50),
	@Address nvarchar(250)
	
AS
    If @mode = 'Add'
	BEGIN
	INSERT INTO tbl_Contact
	(
	[First Name],
	[Last Name],
	Address
	)
	VALUES(
	@FirstName,
	@LastName,
	@Address
		
	)
	End
